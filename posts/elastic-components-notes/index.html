test an apm server #

Make 2 files for running the server and the hey-apm benchmarking tool for it

cat << EOF >> benchmark-apm-server.sh
source ./apm-server.env
$GOPATH/bin/hey-apm -apm-secret ${APMSERVER_TOKEN} -run 60s
EOF

cat << EOF >> start_apm-server.sh

#!/usr/bin/env bash
source ./apm-server.env
$GOPATH/bin/apm-server run -v -e  -c ~/citrix/testing-apm/apm-server.yml -E apm-server.expvar.enabled=true
EOF
